The Complete Python Script

Here is the full, commented, and runnable Python code for the Snakes and Ladders game. This script combines all the logic shown above into a single file.

import random

"""
A simple command-line implementation of the classic Snakes and Ladders game.
Approach:
---------
- The game uses a dictionary to represent snakes and ladders, mapping start positions to end positions.
- The player's position is updated based on dice rolls (randomly generated between 1 and 6).
- If the player lands on a snake or ladder, their position is adjusted accordingly.
- If the player overshoots position 100, they "bounce back" by the excess amount.
- The game loop continues until the player reaches exactly position 100.
- User interaction is handled via input prompts and print statements.
Functions:
----------
- roll_dice(): Simulates rolling a six-sided dice.
- move_player(position, roll): Moves the player, checks for snakes/ladders, and handles overshooting.
- play_game(): Main game loop, manages user interaction and win condition.
Note:
-----
This implementation focuses on clarity and simplicity, using basic Python data structures and control flow.
"""

# Board setup: key is start, value is end (snakes/ladders)
snakes = {16: 6, 47: 26, 49: 11, 56: 53, 62: 19, 64: 60, 87: 24, 93: 73, 95: 75, 98: 78}
ladders = {1: 38, 4: 14, 9: 31, 21: 42, 28: 84, 36: 44, 51: 67, 71: 91, 80: 100}

def roll_dice():
    return random.randint(1, 6)

def move_player(position, roll):
    position += roll
    if position > 100:
        position = 100 - (position - 100) # bounce back if overshoot
    if position in snakes:
        print(f"Oops! Bitten by a snake at {position}. Go down to {snakes[position]}.")
        position = snakes[position]
    elif position in ladders:
        print(f"Yay! Climbed a ladder at {position}. Go up to {ladders[position]}.")
        position = ladders[position]
    return position

def play_game():
    player_pos = 0
    while player_pos < 100:
        input("Press Enter to roll the dice...")
        roll = roll_dice()
        print(f"You rolled a {roll}.")
        player_pos = move_player(player_pos, roll)
        print(f"Your position: {player_pos}\n")
        if player_pos == 100:
            print("Congratulations! You won the game!")

if __name__ == "__main__":
    play_game()
